function muatSyiarIslam() { // 1. Tampilkan Tanggal Masehi Real-time const skrg = new Date(); const opsiMasehi = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' }; document.getElementById('txt-masehi').innerText = "📅 " + skrg.toLocaleDateString('id-ID', opsiMasehi); // 2. 🌟 RUMUS KALKULATOR HIJRIAH OTOMATIS (Akurat Berdasarkan Penanggalan Islam) let masehiWaktu = skrg.getTime(); // Estimasi epoch konstanta kalender Hijriah let hariSejakEpoch = Math.floor(masehiWaktu / 86400000) - 1; let jd = hariSejakEpoch + 2440588; let l = jd - 1948440 + 10632; let n = Math.floor((l - 1) / 10631); l = l - 10631 * n + 354; let j = (Math.floor((10985 - l) / 5316)) * (Math.floor((50 * l) / 17719)) + (Math.floor(l / 5670)) * (Math.floor((43 * l) / 15238)); l = l - (Math.floor((30 - j) / 15)) * (Math.floor((17719 * j) / 50)) - (Math.floor(j / 16)) * (Math.floor((15238 * j) / 43)) + 29; let bulanHijriahKe = Math.floor((24 * l) / 709); let tanggalHijriah = l - Math.floor((709 * bulanHijriahKe) / 24); let tahunHijriah = 30 * n + j - 30; // Daftar Nama-Nama Bulan Islam Resmi const namaBulanIslam = [ "Muharram", "Safar", "Rabi'ul Awwal", "Rabi'ul Akhir", "Jumadil Awwal", "Jumadil Akhir", "Rajab", "Sya'ban", "Ramadhan", "Syawwal", "Dzulqa'dah", "Dzulhijjah" ]; let namaBulanFinal = namaBulanIslam[bulanHijriahKe - 1] || "Dzulhijjah"; // Tampilkan hasil gabungan angka tanggal, nama bulan, dan tahun ke layar HP document.getElementById('txt-hijriah').innerText = `🌙 ${tanggalHijriah} ${namaBulanFinal} ${tahunHijriah} H`; }
